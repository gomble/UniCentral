powershell -Command "iwr '{{ baseUrl }}/api/agent/install-script/windows?key={{ settingsForm.enrollmentKey }}&category=client' -UseBasicParsing | iex"
curl -sL '{{ baseUrl }}/api/agent/install-script/linux?key={{ settingsForm.enrollmentKey }}&category=server' | bash
